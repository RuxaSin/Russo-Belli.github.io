  <script>

    // 🎠 Простая карусель изображений

    document.addEventListener('DOMContentLoaded', function() {

      const track = document.querySelector('.carousel-track');

      const slides = Array.from(track.children);

      const nextBtn = document.querySelector('.carousel-btn.next');

      const prevBtn = document.querySelector('.carousel-btn.prev');

      const indicators = document.querySelectorAll('.indicator');

      let currentIndex = 0;

      let autoPlayInterval;

      // Переход к слайду по индексу

      function goToSlide(index) {

        // Зацикливаем: если вышли за границы — идём с начала/конца

        if (index < 0) index = slides.length - 1;

        if (index >= slides.length) index = 0;

        currentIndex = index;

        // Сдвигаем трек

        track.style.transform = `translateX(-${currentIndex * 100}%)`;

        // Обновляем активный индикатор

        indicators.forEach((ind, i) => {

          ind.classList.toggle('active', i === currentIndex);

        });

      }

      // Проверка: сколько слайдов найдено?

      console.log('Найдено слайдов:', slides.length);

      // Если слайдов нет или один — выходим

      if (slides.length === 0) {

        console.error('❌ Слайды не найдены! Проверь HTML структуру.');

        return;

      }

      // Обработчики кнопок

      if (nextBtn) {

        nextBtn.addEventListener('click', () => {

          goToSlide(currentIndex + 1);

          resetAutoPlay();

        });

      }

      if (prevBtn) {

        prevBtn.addEventListener('click', () => {

          goToSlide(currentIndex - 1);

          resetAutoPlay();

        });

      }

      // Обработчики индикаторов

      indicators.forEach((indicator, index) => {

        indicator.addEventListener('click', () => {

          goToSlide(index);

          resetAutoPlay();

        });

      });

      // Автопрокрутка (опционально — раскомментируй, если нужно)

      function startAutoPlay() {

        autoPlayInterval = setInterval(() => {

          goToSlide(currentIndex + 1);

        }, 5000); // 5 секунд между слайдами

      }

 

      function resetAutoPlay() {

        clearInterval(autoPlayInterval);

        startAutoPlay();

      }

      // Пауза при наведении

      const carousel = document.querySelector('.carousel');

      if (carousel) {

        carousel.addEventListener('mouseenter', () => clearInterval(autoPlayInterval));

        carousel.addEventListener('mouseleave', startAutoPlay);

      }

      // Запуск

      if (slides.length > 1) {

        startAutoPlay();

      }

      // 🎹 Поддержка клавиатуры (стрелки ← →)

      document.addEventListener('keydown', (e) => {

        if (e.key === 'ArrowRight') goToSlide(currentIndex + 1);

        if (e.key === 'ArrowLeft') goToSlide(currentIndex - 1);

      });

    });

  </script>

  <!-- 🖼️ МОДАЛЬНОЕ ОКНО ДЛЯ ЗУМА (ЛАЙТБОКС) -->

  <div id="lightbox" class="lightbox" aria-hidden="true">

    <button class="lightbox-close" aria-label="Закрыть">×</button>

    <img src="" alt="" class="lightbox-image" id="lightbox-img">

    <p class="lightbox-caption" id="lightbox-caption"></p>

  <script>

    // 🖼️ Лайтбокс: зум изображений при клике

    document.addEventListener('DOMContentLoaded', function() {

      const lightbox = document.getElementById('lightbox');

      const lightboxImg = document.getElementById('lightbox-img');

      const lightboxCaption = document.getElementById('lightbox-caption');

      const closeBtn = document.querySelector('.lightbox-close');

      // Находим все изображения с атрибутом data-zoom

      const zoomableImages = document.querySelectorAll('[data-zoom]');

      // Функция открытия лайтбокса

      function openLightbox(src, alt, caption) {

        lightboxImg.src = src;

        lightboxImg.alt = alt;

        lightboxCaption.textContent = caption || '';

        lightbox.classList.add('active');

        lightbox.setAttribute('aria-hidden', 'false');

        document.body.style.overflow = 'hidden'; // Блокируем прокрутку фона

      }

      // Функция закрытия лайтбокса

      function closeLightbox() {

        lightbox.classList.remove('active');

        lightbox.setAttribute('aria-hidden', 'true');

        document.body.style.overflow = ''; // Возвращаем прокрутку

        // Очищаем src после анимации, чтобы не грузить память

        setTimeout(() => {

          lightboxImg.src = '';

        }, 300);

      }

      // Обработчик клика по изображениям

      zoomableImages.forEach(img => {

        img.style.cursor = 'zoom-in'; // Показываем, что можно увеличить

        img.addEventListener('click', function(e) {

          e.stopPropagation(); // Чтобы не сработал клик по карусели

          const zoomSrc = this.getAttribute('data-zoom') || this.src;

          const altText = this.alt;

          const caption = this.getAttribute('data-caption');

          openLightbox(zoomSrc, altText, caption);

        });

      });

      // Закрытие по клику на крестик

      closeBtn.addEventListener('click', closeLightbox);

      // Закрытие по клику на фон (вне изображения)

      lightbox.addEventListener('click', function(e) {

        if (e.target === lightbox) {

          closeLightbox();

        }

      });

      // Закрытие по клавише Escape

      document.addEventListener('keydown', function(e) {

        if (e.key === 'Escape' && lightbox.classList.contains('active')) {

          closeLightbox();

        }

      });

      // 🎹 Поддержка клавиатуры внутри лайтбокса

      lightbox.addEventListener('keydown', function(e) {

        if (e.key === 'Escape') closeLightbox();

      });

    });

  </script>

      // Функция добавления сообщения

      function addMessage(text, sender) {

        const msgDiv = document.createElement('div');

        msgDiv.classList.add('message', sender === 'bot' ? 'bot-msg' : 'user-msg');

        msgDiv.innerHTML = text;

        messagesBox.appendChild(msgDiv);

        messagesBox.scrollTop = messagesBox.scrollHeight; // Автопрокрутка вниз

      }

      // Функция поиска ответа

      function getBotReply(userText) {

        const text = userText.toLowerCase();

        // Ищем совпадение по ключевым словам

        const found = botKnowledge.find(item =>

          item.keywords.some(keyword => text.includes(keyword))

        );

        return found ? found.answer : 'Хм, я пока не знаю ответ на это 🤔. Лучше напишите Руслану напрямую через форму или Telegram!';

      }

      // Обработка отправки

      function sendMessage() {

        const text = input.value.trim();

        if (!text) return;

        addMessage(text, 'user');

        input.value = '';

        // Имитация "думания" бота

        setTimeout(() => {

          const reply = getBotReply(text);

          addMessage(reply, 'bot');

        }, 600); // Задержка 0.6 сек

      }
